select 
v.FK_ERP_ASESORES, 
v.ID,
a.DESCRIPCION,
l.DESCRIPCION,
p.DESCRIPCION,
PT.DESCRIPCION,
rz.RAZON_SOCIAL,
s.CALLE,
s.NUMERO,
s.CP,
s.TE
from erp_com_ven V
join ERP_ASESORES A on v.FK_ERP_ASESORES=a.ID
join ERP_SUCURSALES S on a.FK_ERP_SUCURSALES = s.id
join ERP_LOCALIDADES L on s.FK_ERP_LOCALIDADES = L.id  
join ERP_PAISES P on s.FK_ERP_PAISES = P.ID
join ERP_PARTIDOS PT on s.FK_ERP_PARTIDOS = PT.ID
join ERP_RAZ_SOCIALES RZ on s.FK_ERP_RAZ_SOCIALES = RZ.ID
