Select 
A.ID,
A.DESCRIPCION,
AG.DESCRIPCION AS AGRUPACION,
A.FK_ERP_TIP_ART,
A.COMISION,
PR.DESCRIPCION AS PRESENTACION,
(select SUM(S.SAL_COMP_ING) 
from
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 1 ) AS 'SAL_PEN_ING',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 1) AS 'DEP_CENTRAL',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 2) AS 'COLCHONES',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 3) AS 'MUEBLES',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 4) AS 'SALON_FRENTE',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 5) AS 'SALON',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 6) AS 'ELECTRO',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 7) AS 'OUTLET',
(select SUM(S.SAL_REAL) 
from 
_MR_SAL_STO S
where
S.FK_ERP_ARTICULOS = A.ID and
S.FK_ERP_DEPOSITOS = 8) AS 'SERV_TEC',
A.HABILITADO,
A.VENTA,
P.FEC_ULT_ACT,
L.ID AS   FK_ERP_LIS_PRECIO,
(P.PRECIO / 1.1) AS PRECIOSDESC,

(select (PRECIO) from ERP_PRE_VEN PRE where FK_ERP_ARTICULOS = A.ID and FK_ERP_LIS_PRECIO= '2' ) AS PRECIO

from ERP_ARTICULOS A
left join ERP_AGR_ART AG on AG.ID = A.FK_ERP_AGR_ART
left join ERP_PRE_ART PR on PR.ID = A.FK_ERP_PRE_ART
join ERP_LIS_PRECIO L on 1=1
left join ERP_PRE_VEN P on P.FK_ERP_ARTICULOS = A.ID  and P.FK_ERP_LIS_PRECIO = L.ID

where 
A.HABILITADO = 1 and
A.VENTA = 1 and L.ID = 2
